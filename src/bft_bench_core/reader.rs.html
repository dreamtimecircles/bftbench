1use std::time::{Duration, Instant};
2
3use tokio::sync::{broadcast, mpsc};
4use uuid::Uuid;
5
6use crate::{BftError, BftReader, worker::WorkerRequest};
7
8#[derive(Debug)]
9pub(crate) enum ReaderReply {
10    SuccessfulRead {
11        read_completion_instant: Instant,
12        read_duration: Duration,
13        uuid: Uuid,
14        node_idx: usize,
15    },
16    FailedRead {
17        read_duration: Duration,
18        bft_error: BftError,
19        node_idx: usize,
20    },
21    Completed {
22        node_idx: usize,
23    },
24}
25
26pub(crate) async fn read<R: BftReader + 'static>(
27    node_idx: usize,
28    mut reader: R,
29    mut rx_readers_control: broadcast::Receiver<WorkerRequest>,
30    tx_incoming_reads: mpsc::Sender<ReaderReply>,
31    read_grace: Duration,
32) {
33    loop {
34        let read_start = Instant::now();
35        log::debug!("Reader {}: reading", node_idx);
36        match tokio::time::timeout(read_grace, reader.read()).await {
37            Ok(read_result) => {
38                let read_completion_instant = Instant::now();
39                let read_duration = read_start.elapsed();
40                match read_result {
41                    Ok(Some(uuid)) => {
42                        log::debug!("Reader {}: read transaction {}", node_idx, uuid);
43                        tx_incoming_reads
44                            .send(ReaderReply::SuccessfulRead {
45                                read_completion_instant,
46                                read_duration,
47                                uuid,
48                                node_idx,
49                            })
50                            .await
51                            .expect("Receiver closed");
52                    }
53                    Ok(None) => {
54                        log::debug!("Reader {}: ending (closed from reader)", node_idx);
55                        tx_incoming_reads
56                            .send(ReaderReply::Completed { node_idx })
57                            .await
58                            .expect("Reader completion message couldn't be sent");
59                        break;
60                    }
61                    Err(bft_error) => {
62                        log::error!("Reader {}: read failed: {}", node_idx, bft_error);
63                        tx_incoming_reads
64                            .send(ReaderReply::FailedRead {
65                                read_duration,
66                                node_idx,
67                                bft_error,
68                            })
69                            .await
70                            .expect("Receiver closed");
71                    }
72                }
73            }
74            Err(_) => {
75                if let Ok(WorkerRequest::Stop()) = rx_readers_control.try_recv() {
76                    log::debug!("Reader {}: ending (closed by completion command)", node_idx);
77                    tx_incoming_reads
78                        .send(ReaderReply::Completed { node_idx })
79                        .await
80                        .expect("Reader completion message couldn't be sent");
81                    break;
82                } else {
83                    log::warn!("Reader {}: timeout", node_idx);
84                }
85            }
86        }
87    }
88}
