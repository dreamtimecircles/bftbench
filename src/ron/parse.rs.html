17// We have the following char categories.
18const INT_CHAR: u8 = 1 << 0; // [0-9A-Fa-f_]
19const FLOAT_CHAR: u8 = 1 << 1; // [0-9\.Ee+-_]
20const IDENT_FIRST_CHAR: u8 = 1 << 2; // [A-Za-z_]
21const IDENT_OTHER_CHAR: u8 = 1 << 3; // [A-Za-z_0-9]
22const IDENT_RAW_CHAR: u8 = 1 << 4; // [A-Za-z_0-9\.+-]
23const WHITESPACE_CHAR: u8 = 1 << 5; // [\n\t\r ]
24
25// We encode each char as belonging to some number of these categories.
26const DIGIT: u8 = INT_CHAR | FLOAT_CHAR | IDENT_OTHER_CHAR | IDENT_RAW_CHAR; // [0-9]
27const ABCDF: u8 = INT_CHAR | IDENT_FIRST_CHAR | IDENT_OTHER_CHAR | IDENT_RAW_CHAR; // [ABCDFabcdf]
28const UNDER: u8 = INT_CHAR | FLOAT_CHAR | IDENT_FIRST_CHAR | IDENT_OTHER_CHAR | IDENT_RAW_CHAR; // [_]
29const E____: u8 = INT_CHAR | FLOAT_CHAR | IDENT_FIRST_CHAR | IDENT_OTHER_CHAR | IDENT_RAW_CHAR; // [Ee]
30const G2Z__: u8 = IDENT_FIRST_CHAR | IDENT_OTHER_CHAR | IDENT_RAW_CHAR; // [G-Zg-z]
31const PUNCT: u8 = FLOAT_CHAR | IDENT_RAW_CHAR; // [\.+-]
32const WS___: u8 = WHITESPACE_CHAR; // [\t\n\r ]
33const _____: u8 = 0; // everything else

128    pub fn new(bytes: &'a [u8]) -> SpannedResult<Self> {
129        let mut b = Bytes {
130            exts: Extensions::empty(),
131            bytes,
132            cursor: Position { line: 1, col: 1 },
133        };
134
135        b.skip_ws().map_err(|e| b.span_error(e))?;
136
137        // Loop over all extensions attributes
138        loop {
139            let attribute = b.extensions().map_err(|e| b.span_error(e))?;
140
141            if attribute.is_empty() {
142                break;
143            }
144
145            b.exts |= attribute;
146            b.skip_ws().map_err(|e| b.span_error(e))?;
147        }
148
149        Ok(b)
150    }

167    pub fn advance_single(&mut self) -> Result<()> {
168        if self.peek_or_eof()? == b'\n' {
169            self.cursor.line += 1;
170            self.cursor.col = 1;
171        } else {
172            self.cursor.col += 1;
173        }
174
175        self.bytes = &self.bytes[1..];
176
177        Ok(())

180    fn any_integer<T: Num>(&mut self, sign: i8) -> Result<T> {
181        let base = if self.peek() == Some(b'0') {
182            match self.bytes.get(1).copied() {
183                Some(b'x') => 16,
184                Some(b'b') => 2,
185                Some(b'o') => 8,
186                _ => 10,
187            }
188        } else {
189            10
190        };
191
192        if base != 10 {
193            // If we have `0x45A` for example,
194            // cut it to `45A`.
195            let _ = self.advance(2);
196        }
197
198        let num_bytes = self.next_bytes_contained_in(is_int_char);
199
200        if num_bytes == 0 {
201            return Err(Error::ExpectedInteger);
202        }
203
204        let s = unsafe { from_utf8_unchecked(&self.bytes[0..num_bytes]) };
205
206        if s.as_bytes()[0] == b'_' {
207            return Err(Error::UnderscoreAtBeginning);
208        }
209
210        fn calc_num<T: Num>(
211            bytes: &Bytes,
212            s: &str,
213            base: u8,
214            mut f: impl FnMut(&mut T, u8) -> bool,
215        ) -> Result<T> {
216            let mut num_acc = T::from_u8(0);
217
218            for &byte in s.as_bytes() {
219                if byte == b'_' {
220                    continue;
221                }
222
223                if num_acc.checked_mul_ext(base) {
224                    return Err(Error::IntegerOutOfBounds);
225                }
226
227                let digit = bytes.decode_hex(byte)?;
228
229                if digit >= base {
230                    return Err(Error::ExpectedInteger);
231                }
232
233                if f(&mut num_acc, digit) {
234                    return Err(Error::IntegerOutOfBounds);
235                }
236            }
237
238            Ok(num_acc)
239        }
240
241        let res = if sign > 0 {
242            calc_num(self, s, base, T::checked_add_ext)
243        } else {
244            calc_num(self, s, base, T::checked_sub_ext)
245        };
246
247        let _ = self.advance(num_bytes);
248
249        res
250    }

252    pub fn any_num(&mut self) -> Result<AnyNum> {
253        // We are not doing float comparisons here in the traditional sense.
254        // Instead, this code checks if a f64 fits inside an f32.
255        #[allow(clippy::float_cmp)]
256        fn any_float(f: f64) -> Result<AnyNum> {
257            if f == f64::from(f as f32) {
258                Ok(AnyNum::F32(f as f32))
259            } else {
260                Ok(AnyNum::F64(f))
261            }
262        }
263
264        let bytes_backup = self.bytes;
265
266        let first_byte = self.peek_or_eof()?;
267        let is_signed = first_byte == b'-' || first_byte == b'+';
268        let is_float = self.next_bytes_is_float();
269
270        if is_float {
271            let f = self.float::<f64>()?;
272
273            any_float(f)
274        } else {
275            let max_u8 = LargeUInt::from(std::u8::MAX);
276            let max_u16 = LargeUInt::from(std::u16::MAX);
277            let max_u32 = LargeUInt::from(std::u32::MAX);
278            #[cfg_attr(not(feature = "integer128"), allow(clippy::useless_conversion))]
279            let max_u64 = LargeUInt::from(std::u64::MAX);
280
281            let min_i8 = LargeSInt::from(std::i8::MIN);
282            let max_i8 = LargeSInt::from(std::i8::MAX);
283            let min_i16 = LargeSInt::from(std::i16::MIN);
284            let max_i16 = LargeSInt::from(std::i16::MAX);
285            let min_i32 = LargeSInt::from(std::i32::MIN);
286            let max_i32 = LargeSInt::from(std::i32::MAX);
287            #[cfg_attr(not(feature = "integer128"), allow(clippy::useless_conversion))]
288            let min_i64 = LargeSInt::from(std::i64::MIN);
289            #[cfg_attr(not(feature = "integer128"), allow(clippy::useless_conversion))]
290            let max_i64 = LargeSInt::from(std::i64::MAX);
291
292            if is_signed {
293                match self.signed_integer::<LargeSInt>() {
294                    Ok(x) => {
295                        if x >= min_i8 && x <= max_i8 {
296                            Ok(AnyNum::I8(x as i8))
297                        } else if x >= min_i16 && x <= max_i16 {
298                            Ok(AnyNum::I16(x as i16))
299                        } else if x >= min_i32 && x <= max_i32 {
300                            Ok(AnyNum::I32(x as i32))
301                        } else if x >= min_i64 && x <= max_i64 {
302                            Ok(AnyNum::I64(x as i64))
303                        } else {
304                            #[cfg(feature = "integer128")]
305                            {
306                                Ok(AnyNum::I128(x))
307                            }
308                            #[cfg(not(feature = "integer128"))]
309                            {
310                                Ok(AnyNum::I64(x))
311                            }
312                        }
313                    }
314                    Err(_) => {
315                        self.bytes = bytes_backup;
316
317                        any_float(self.float::<f64>()?)
318                    }
319                }
320            } else {
321                match self.unsigned_integer::<LargeUInt>() {
322                    Ok(x) => {
323                        if x <= max_u8 {
324                            Ok(AnyNum::U8(x as u8))
325                        } else if x <= max_u16 {
326                            Ok(AnyNum::U16(x as u16))
327                        } else if x <= max_u32 {
328                            Ok(AnyNum::U32(x as u32))
329                        } else if x <= max_u64 {
330                            Ok(AnyNum::U64(x as u64))
331                        } else {
332                            #[cfg(feature = "integer128")]
333                            {
334                                Ok(AnyNum::U128(x))
335                            }
336                            #[cfg(not(feature = "integer128"))]
337                            {
338                                Ok(AnyNum::U64(x))
339                            }
340                        }
341                    }
342                    Err(_) => {
343                        self.bytes = bytes_backup;
344
345                        any_float(self.float::<f64>()?)
346                    }
347                }
348            }
349        }
350    }

366    pub fn char(&mut self) -> Result<char> {
367        if !self.consume("'") {
368            return Err(Error::ExpectedChar);
369        }
370
371        let c = self.peek_or_eof()?;
372
373        let c = if c == b'\\' {
374            let _ = self.advance(1);
375
376            self.parse_escape()?
377        } else {
378            // Check where the end of the char (') is and try to
379            // interpret the rest as UTF-8
380
381            let max = self.bytes.len().min(5);
382            let pos: usize = self.bytes[..max]
383                .iter()
384                .position(|&x| x == b'\'')
385                .ok_or(Error::ExpectedChar)?;
386            let s = from_utf8(&self.bytes[0..pos]).map_err(Error::from)?;
387            let mut chars = s.chars();
388
389            let first = chars.next().ok_or(Error::ExpectedChar)?;
390            if chars.next().is_some() {
391                return Err(Error::ExpectedChar);
392            }
393
394            let _ = self.advance(pos);
395
396            first
397        };
398
399        if !self.consume("'") {
400            return Err(Error::ExpectedChar);
401        }
402
403        Ok(c)
404    }

455    pub fn consume_struct_name(&mut self, ident: &'static str) -> Result<bool> {
456        if self.check_ident("") {
457            return Ok(false);
458        }
459
460        let found_ident = match self.identifier() {
461            Ok(maybe_ident) => std::str::from_utf8(maybe_ident)?,
462            Err(Error::SuggestRawIdentifier(found_ident)) if found_ident == ident => {
463                return Err(Error::SuggestRawIdentifier(found_ident))
464            }
465            Err(_) => return Err(Error::ExpectedNamedStructLike(ident)),
466        };
467
468        if found_ident != ident {
469            return Err(Error::ExpectedDifferentStructName {
470                expected: ident,
471                found: String::from(found_ident),
472            });
473        }
474
475        Ok(true)
476    }

488    fn consume_all(&mut self, all: &[&str]) -> Result<bool> {
489        all.iter()
490            .map(|elem| {
491                if self.consume(elem) {
492                    self.skip_ws()?;
493
494                    Ok(true)
495                } else {
496                    Ok(false)
497                }
498            })
499            .fold(Ok(true), |acc, x| acc.and_then(|val| x.map(|x| x && val)))
500    }

514    /// Returns the extensions bit mask.
515    fn extensions(&mut self) -> Result<Extensions> {
516        if self.peek() != Some(b'#') {
517            return Ok(Extensions::empty());
518        }
519
520        if !self.consume_all(&["#", "!", "[", "enable", "("])? {
521            return Err(Error::ExpectedAttribute);
522        }
523
524        self.skip_ws()?;
525        let mut extensions = Extensions::empty();
526
527        loop {
528            let ident = self.identifier()?;
529            let extension = Extensions::from_ident(ident).ok_or_else(|| {
530                Error::NoSuchExtension(String::from_utf8_lossy(ident).into_owned())
531            })?;
532
533            extensions |= extension;
534
535            let comma = self.comma()?;
536
537            // If we have no comma but another item, return an error
538            if !comma && self.check_ident_other_char(0) {
539                return Err(Error::ExpectedComma);
540            }
541
542            // If there's no comma, assume the list ended.
543            // If there is, it might be a trailing one, thus we only
544            // continue the loop if we get an ident char.
545            if !comma || !self.check_ident_other_char(0) {
546                break;
547            }
548        }
549
550        self.skip_ws()?;
551
552        if self.consume_all(&[")", "]"])? {
553            Ok(extensions)
554        } else {
555            Err(Error::ExpectedAttributeEnd)
556        }
557    }

559    pub fn float<T>(&mut self) -> Result<T>
560    where
561        T: FromStr,
562    {
563        for literal in &["inf", "+inf", "-inf", "NaN", "+NaN", "-NaN"] {
564            if self.consume_ident(literal) {
565                return FromStr::from_str(literal).map_err(|_| unreachable!()); // must not fail
566            }
567        }
568
569        let num_bytes = self.next_bytes_contained_in(is_float_char);
570
571        // Since `rustc` allows `1_0.0_1`, lint against underscores in floats
572        if let Some(err_bytes) = self.bytes[0..num_bytes].iter().position(|b| *b == b'_') {
573            let _ = self.advance(err_bytes);
574
575            return Err(Error::FloatUnderscore);
576        }
577
578        let s = unsafe { from_utf8_unchecked(&self.bytes[0..num_bytes]) };
579        let res = FromStr::from_str(s).map_err(|_| Error::ExpectedFloat);
580
581        let _ = self.advance(num_bytes);
582
583        res
584    }

586    pub fn identifier(&mut self) -> Result<&'a [u8]> {
587        let next = self.peek_or_eof()?;
588        if !is_ident_first_char(next) {
589            if is_ident_raw_char(next) {
590                let ident_bytes = &self.bytes[..self.next_bytes_contained_in(is_ident_raw_char)];
591
592                if let Ok(ident) = std::str::from_utf8(ident_bytes) {
593                    return Err(Error::SuggestRawIdentifier(String::from(ident)));
594                }
595            }
596
597            return Err(Error::ExpectedIdentifier);
598        }
599
600        // If the next two bytes signify the start of a raw string literal,
601        // return an error.
602        let length = if next == b'r' {
603            match self.bytes.get(1).ok_or(Error::Eof)? {
604                b'"' => return Err(Error::ExpectedIdentifier),
605                b'#' => {
606                    let after_next = self.bytes.get(2).copied().unwrap_or_default();
607                    // Note: it's important to check this before advancing forward, so that
608                    // the value-type deserializer can fall back to parsing it differently.
609                    if !is_ident_raw_char(after_next) {
610                        return Err(Error::ExpectedIdentifier);
611                    }
612                    // skip "r#"
613                    let _ = self.advance(2);
614                    self.next_bytes_contained_in(is_ident_raw_char)
615                }
616                _ => {
617                    let std_ident_length = self.next_bytes_contained_in(is_ident_other_char);
618                    let raw_ident_length = self.next_bytes_contained_in(is_ident_raw_char);
619
620                    if raw_ident_length > std_ident_length {
621                        if let Ok(ident) = std::str::from_utf8(&self.bytes[..raw_ident_length]) {
622                            return Err(Error::SuggestRawIdentifier(String::from(ident)));
623                        }
624                    }
625
626                    std_ident_length
627                }
628            }
629        } else {
630            let std_ident_length = self.next_bytes_contained_in(is_ident_other_char);
631            let raw_ident_length = self.next_bytes_contained_in(is_ident_raw_char);
632
633            if raw_ident_length > std_ident_length {
634                if let Ok(ident) = std::str::from_utf8(&self.bytes[..raw_ident_length]) {
635                    return Err(Error::SuggestRawIdentifier(String::from(ident)));
636                }
637            }
638
639            std_ident_length
640        };
641
642        let ident = &self.bytes[..length];
643        let _ = self.advance(length);
644
645        Ok(ident)
646    }

648    pub fn next_bytes_contained_in(&self, allowed: fn(u8) -> bool) -> usize {
649        self.bytes.iter().take_while(|&&b| allowed(b)).count()
650    }

652    pub fn next_bytes_is_float(&self) -> bool {
653        if let Some(byte) = self.peek() {
654            let skip = match byte {
655                b'+' | b'-' => 1,
656                _ => 0,
657            };
658            let flen = self
659                .bytes
660                .iter()
661                .skip(skip)
662                .take_while(|&&b| is_float_char(b))
663                .count();
664            let ilen = self
665                .bytes
666                .iter()
667                .skip(skip)
668                .take_while(|&&b| is_int_char(b))
669                .count();
670            flen > ilen
671        } else {
672            false
673        }
674    }

679                let _ = self.advance_single();
680            }

682            if !self.skip_comment()? {
683                return Ok(());
684            }
685        }

696    pub fn signed_integer<T>(&mut self) -> Result<T>
697    where
698        T: Num,
699    {
700        match self.peek_or_eof()? {
701            b'+' => {
702                let _ = self.advance_single();
703
704                self.any_integer(1)
705            }
706            b'-' => {
707                let _ = self.advance_single();

715    pub fn string(&mut self) -> Result<ParsedStr<'a>> {
716        if self.consume("\"") {
717            self.escaped_string()
718        } else if self.consume("r") {
719            self.raw_string()
720        } else {
721            Err(Error::ExpectedString)
722        }
723    }
724
725    fn escaped_string(&mut self) -> Result<ParsedStr<'a>> {
726        use std::iter::repeat;
727
728        let (i, end_or_escape) = self
729            .bytes
730            .iter()
731            .enumerate()
732            .find(|&(_, &b)| b == b'\\' || b == b'"')
733            .ok_or(Error::ExpectedStringEnd)?;
734
735        if *end_or_escape == b'"' {
736            let s = from_utf8(&self.bytes[..i]).map_err(Error::from)?;
737
738            // Advance by the number of bytes of the string
739            // + 1 for the `"`.
740            let _ = self.advance(i + 1);
741
742            Ok(ParsedStr::Slice(s))
743        } else {
744            let mut i = i;
745            let mut s: Vec<_> = self.bytes[..i].to_vec();
746
747            loop {
748                let _ = self.advance(i + 1);
749                let character = self.parse_escape()?;
750                match character.len_utf8() {
751                    1 => s.push(character as u8),
752                    len => {
753                        let start = s.len();
754                        s.extend(repeat(0).take(len));
755                        character.encode_utf8(&mut s[start..]);
756                    }
757                }
758
759                let (new_i, end_or_escape) = self
760                    .bytes
761                    .iter()
762                    .enumerate()
763                    .find(|&(_, &b)| b == b'\\' || b == b'"')
764                    .ok_or(Error::ExpectedStringEnd)?;
765
766                i = new_i;
767                s.extend_from_slice(&self.bytes[..i]);
768
769                if *end_or_escape == b'"' {
770                    let _ = self.advance(i + 1);
771
772                    let s = String::from_utf8(s).map_err(Error::from)?;
773                    break Ok(ParsedStr::Allocated(s));
774                }
775            }
776        }
777    }
778
779    fn raw_string(&mut self) -> Result<ParsedStr<'a>> {
780        let num_hashes = self.bytes.iter().take_while(|&&b| b == b'#').count();
781        let hashes = &self.bytes[..num_hashes];
782        let _ = self.advance(num_hashes);
783
784        if !self.consume("\"") {
785            return Err(Error::ExpectedString);
786        }
787
788        let ending = [&[b'"'], hashes].concat();
789        let i = self
790            .bytes
791            .windows(num_hashes + 1)
792            .position(|window| window == ending.as_slice())
793            .ok_or(Error::ExpectedStringEnd)?;
794
795        let s = from_utf8(&self.bytes[..i]).map_err(Error::from)?;
796
797        // Advance by the number of bytes of the string
798        // + `num_hashes` + 1 for the `"`.
799        let _ = self.advance(i + num_hashes + 1);
800
801        Ok(ParsedStr::Slice(s))
802    }

804    fn test_for(&self, s: &str) -> bool {
805        s.bytes()
806            .enumerate()
807            .all(|(i, b)| self.bytes.get(i).map_or(false, |t| *t == b))
808    }

810    pub fn unsigned_integer<T: Num>(&mut self) -> Result<T> {
811        self.any_integer(1)
812    }

814    fn decode_ascii_escape(&mut self) -> Result<u8> {
815        let mut n = 0;
816        for _ in 0..2 {
817            n <<= 4;
818            let byte = self.eat_byte()?;
819            let decoded = self.decode_hex(byte)?;
820            n |= decoded;
821        }
822
823        Ok(n)
824    }
825
826    #[inline]
827    fn decode_hex(&self, c: u8) -> Result<u8> {
828        match c {
829            c @ b'0'..=b'9' => Ok(c - b'0'),
830            c @ b'a'..=b'f' => Ok(10 + c - b'a'),
831            c @ b'A'..=b'F' => Ok(10 + c - b'A'),
832            _ => Err(Error::InvalidEscape("Non-hex digit found")),
833        }
834    }
835
836    fn parse_escape(&mut self) -> Result<char> {
837        let c = match self.eat_byte()? {
838            b'\'' => '\'',
839            b'"' => '"',
840            b'\\' => '\\',
841            b'n' => '\n',
842            b'r' => '\r',
843            b't' => '\t',
844            b'0' => '\0',
845            b'x' => self.decode_ascii_escape()? as char,
846            b'u' => {
847                self.expect_byte(b'{', Error::InvalidEscape("Missing { in Unicode escape"))?;
848
849                let mut bytes: u32 = 0;
850                let mut num_digits = 0;
851
852                while num_digits < 6 {
853                    let byte = self.peek_or_eof()?;
854
855                    if byte == b'}' {
856                        break;
857                    } else {
858                        self.advance_single()?;
859                    }
860
861                    let byte = self.decode_hex(byte)?;
862                    bytes <<= 4;
863                    bytes |= u32::from(byte);
864
865                    num_digits += 1;
866                }
867
868                if num_digits == 0 {
869                    return Err(Error::InvalidEscape(
870                        "Expected 1-6 digits, got 0 digits in Unicode escape",
871                    ));
872                }
873
874                self.expect_byte(
875                    b'}',
876                    Error::InvalidEscape("No } at the end of Unicode escape"),
877                )?;
878                char_from_u32(bytes).ok_or(Error::InvalidEscape("Not a valid char"))?
879            }
880            _ => {
881                return Err(Error::InvalidEscape("Unknown escape character"));
882            }
883        };
884
885        Ok(c)
886    }
887
888    fn skip_comment(&mut self) -> Result<bool> {
889        if self.consume("/") {
890            match self.eat_byte()? {
891                b'/' => {
892                    let bytes = self.bytes.iter().take_while(|&&b| b != b'\n').count();
893
894                    let _ = self.advance(bytes);
895                }
896                b'*' => {
897                    let mut level = 1;
898
899                    while level > 0 {
900                        let bytes = self
901                            .bytes
902                            .iter()
903                            .take_while(|&&b| b != b'/' && b != b'*')
904                            .count();
905
906                        if self.bytes.is_empty() {
907                            return Err(Error::UnclosedBlockComment);
908                        }
909
910                        let _ = self.advance(bytes);
911
912                        // check whether / or * and take action
913                        if self.consume("/*") {
914                            level += 1;
915                        } else if self.consume("*/") {
916                            level -= 1;
917                        } else {
918                            self.eat_byte().map_err(|_| Error::UnclosedBlockComment)?;
919                        }
920                    }
921                }
922                b => return Err(Error::UnexpectedByte(b as char)),
923            }
924
925            Ok(true)
926        } else {
927            Ok(false)
928        }
929    }
930}

932pub trait Num {
933    fn from_u8(x: u8) -> Self;
934
935    /// Returns `true` on overflow
936    fn checked_mul_ext(&mut self, x: u8) -> bool;
937
938    /// Returns `true` on overflow
939    fn checked_add_ext(&mut self, x: u8) -> bool;
940
941    /// Returns `true` on overflow
942    fn checked_sub_ext(&mut self, x: u8) -> bool;
943}
944
945macro_rules! impl_num {
946    ($ty:ident) => {
947        impl Num for $ty {
948            fn from_u8(x: u8) -> Self {
949                x as $ty
950            }
951
952            fn checked_mul_ext(&mut self, x: u8) -> bool {
953                match self.checked_mul(Self::from_u8(x)) {
954                    Some(n) => {
955                        *self = n;
956                        false
957                    }
958                    None => true,
959                }
960            }
961
962            fn checked_add_ext(&mut self, x: u8) -> bool {
963                match self.checked_add(Self::from_u8(x)) {
964                    Some(n) => {
965                        *self = n;
966                        false
967                    }
968                    None => true,
969                }
970            }
971
972            fn checked_sub_ext(&mut self, x: u8) -> bool {
973                match self.checked_sub(Self::from_u8(x)) {
974                    Some(n) => {
975                        *self = n;
976                        false
977                    }
978                    None => true,
979                }
980            }
981        }
982    };
983    ($($tys:ident)*) => {
984        $( impl_num!($tys); )*
985    };
986}
